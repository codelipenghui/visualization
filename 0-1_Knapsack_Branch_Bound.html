def branch_and_bound(items, capacity):
priority_queue = [initial_node] # (UB, value, weight, level, path)
global_best_value = 0
while priority_queue:
current_node = priority_queue.pop() # Using LIFO for DFS-like behavior
ub, val, wt, lvl, path = current_node
if ub <= global_best_value: # Pruning by bound
continue
if lvl == N_items: # Leaf node
if val > global_best_value:
global_best_value = val
continue
# Branch 1: Exclude item items[lvl]
node_exclude = create_node(val, wt, lvl + 1, path, excluded=True)
if node_exclude.ub > global_best_value:
priority_queue.append(node_exclude)
# Branch 2: Include item items[lvl]
if wt + items[lvl].weight <= W_capacity:
node_include = create_node(val_incl, wt_incl, lvl + 1, path_incl, included=True)
if node_include.ub > global_best_value:
priority_queue.append(node_include)
return global_best_value
