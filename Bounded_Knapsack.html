def bounded_knapsack_dp(v, w, counts, C):
n_types = len(v)
dp = [[0]*(C+1) for _ in range(n_types+1)]
for i in range(1, n_types + 1):
item_v = v[i-1]; item_w = w[i-1]; item_count = counts[i-1]
for cap in range(1, C + 1):
dp[i][cap] = dp[i-1][cap] # Option: Don't take any of current item type
for k in range(1, item_count + 1):
if cap >= k * item_w:
val_with_k = dp[i-1][cap - k * item_w] + k * item_v
dp[i][cap] = max(dp[i][cap], val_with_k)
else:
break # Not enough capacity for k items of this type
return dp[n_types][C]
