def fractional_knapsack(items, capacity):
# items = list of (value, weight, original_id) tuples
items.sort(key=lambda x: x[0]/x[1], reverse=True)
total_value = 0
current_weight = 0
knapsack_items = []
for value, weight, item_id in items:
if current_weight + weight <= capacity:
# Take the whole item
current_weight += weight
total_value += value
knapsack_items.append((item_id, 1.0)) # 1.0 means 100%
else:
# Take a fraction of the item
remaining_capacity = capacity - current_weight
fraction = remaining_capacity / weight
total_value += value * fraction
current_weight += weight * fraction
knapsack_items.append((item_id, fraction))
break # Knapsack is full
return total_value, knapsack_items
