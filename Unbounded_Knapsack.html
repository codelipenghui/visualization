def unbounded_knapsack_dp(v, w, C):
n = len(v)
# dp[i][cap] = max value using first i types of items, with capacity cap
dp = [[0]*(C+1) for _ in range(n+1)]
for i in range(1, n + 1):
for cap in range(1, C + 1):
item_w = w[i-1]; item_v = v[i-1]
# Option 1: Exclude item type i
val_exclude = dp[i-1][cap]
val_include = 0
# Option 2: Include item type i (if it fits)
if cap >= item_w:
# Note: dp[i][...] because item can be chosen multiple times
val_include = item_v + dp[i][cap - item_w]
dp[i][cap] = max(val_exclude, val_include if cap >= item_w else 0)
return dp[n][C]
